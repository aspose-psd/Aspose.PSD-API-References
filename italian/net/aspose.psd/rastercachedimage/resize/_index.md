```csharp
[C#]

string sourceFile = "sample.psd";
string destName = "ResamplerSinCStripes_after.psd";

// Carica un'immagine esistente in un'istanza della classe PsdImage
using (PsdImage image = (PsdImage)Image.Load(sourceFile))
```
